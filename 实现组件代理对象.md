如何通过 this 访问 setup 上定义的属性
`example/helloworld/App.js`
```js
import { h } from '../../lib/mini-vue.esm.js'
export default App = {
	render(){
		return h('div', {id: 'root'}, 'hi ' + this.msg)
	},
	setup(){
		return {
			msg: 'mini-vue'
		}
	}
}
```
